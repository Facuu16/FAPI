var ChatColor = Java.type("org.bukkit.ChatColor");

function onInit() {
    var managerId = "FAPI-MANAGER-STRING";
    
    API.setTempData(managerId, getManager());
}

function getManager() {
    return {
        setColor: setColor
    };
}

/**
 * @function setColor
 * @description Translates alternate color codes in the given string to their corresponding ChatColor codes.
 * 
 * @param {String} string - The input string to translate.
 * 
 * @returns {String} - The translated string with ChatColor codes.
 */
function setColor(string) {
    return ChatColor.translateAlternateColorCodes("&", string);
}
