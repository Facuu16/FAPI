function onInit() {
    var managerId = "FAPI-MANAGER-ARRAY";
    
    API.setTempData(managerId, getManager());
}

function getManager() {
    return {
        contains: contains,
        get: get,
        remove: remove
    };
}

/**
 * @function contains
 * @description Checks if an object is in an array
 * 
 * @param {Object} object - Object to search for in the array
 * @param {Array} array - Array to search the object in
 * 
 * @returns {Boolean} True if the object is in the array, false otherwise
 */
function contains(object, array) {
    return array.indexOf(object) != -1;
}

/**
 * @function get
 * @description Gets an object from an array
 * 
 * @param {Object|Number} object - Object or index to get from the array
 * @param {Array} array - Array to get the object from
 * 
 * @returns {Object} The object if it is found, undefined otherwise
 */
function get(object, array) {
    if (Number.isInteger(object)) 
        return array[object];

    return array[array.indexOf(object)];
}

/**
 * @function remove
 * @description Removes an object from an array
 * 
 * @param {Object} object - Object to remove from the array
 * @param {Array} array - Array to remove the object from
 * 
 * @returns {Array} The array with the object removed
 */
function remove(object, array) {
    return array.filter(function(element) {
        return element !== object;
    });
}
