function onInit() {
    var managerId = "FAPI-MANAGER-PLAYER";
    
    API.setTempData(managerId, getManager());
}

function getManager() {
    return {
        hasArmor: hasArmor
    };
}

/**
 * @function hasArmor
 * @description Checks if a player has the specified armor equipped.
 * 
 * @param {Player} player - The player to check the armor for.
 * @param {String} helmet - The ID of the helmet to check for.
 * @param {String} chestplate - The ID of the chestplate to check for.
 * @param {String} leggings - The ID of the leggings to check for.
 * @param {String} boots - The ID of the boots to check for.
 * 
 * @returns {Boolean} - Returns true if the player has the specified armor equipped, or false if not.
 */
function hasArmor(player, helmet, chestplate, leggings, boots) {
    try {
        var inventory = player.getInventory();

        var armor = {
            player: [
                inventory.getHelmet(), 
                inventory.getChestplate(), 
                inventory.getLeggings(), 
                inventory.getBoots()
            ],
            needed: [
                helmet,
                chestplate,
                leggings,
                boots
            ]
        }

        for (var i = 0; i < 4; i++) {
            if (armor.needed[i] == "none")
                continue;
            
            if (armor.player[i] == null || armor.player[i].getTypeId() != armor.needed[i])
                return false;
        }

        return true;
    } catch (exception) {
        API.message(player, "&c¡Ocurrió un error, por favor reportalo en un ticket! &7{FAPI:HAS_ARMOR}");
        return false;
    }
}
